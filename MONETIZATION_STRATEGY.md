## 8. Feature Gating Implementation

### Database Schema for Subscriptions
```sql
CREATE TABLE subscriptions (
  id VARCHAR(255) PRIMARY KEY,
  project_id VARCHAR(255) NOT NULL,
  tier ENUM('starter', 'professional', 'enterprise') DEFAULT 'starter',
  stripe_subscription_id VARCHAR(255),
  status ENUM('active', 'cancelled', 'past_due') DEFAULT 'active',
  current_period_start TIMESTAMP,
  current_period_end TIMESTAMP,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

CREATE TABLE usage_metrics (
  id VARCHAR(255) PRIMARY KEY,
  project_id VARCHAR(255) NOT NULL,
  metric_type VARCHAR(100),
  value INT,
  timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE add_ons (
  id VARCHAR(255) PRIMARY KEY,
  subscription_id VARCHAR(255) NOT NULL,
  add_on_type VARCHAR(100),
  stripe_product_id VARCHAR(255),
  status ENUM('active', 'inactive') DEFAULT 'active',
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
```

### Feature Access Control
```typescript
// Check if feature is available for tier
async function hasFeatureAccess(projectId: string, feature: string): Promise<boolean> {
  const subscription = await getSubscription(projectId);
  const tierFeatures = getTierFeatures(subscription.tier);
  return tierFeatures.includes(feature);
}

// Check usage limits
async function checkUsageLimit(projectId: string, metric: string): Promise<boolean> {
  const subscription = await getSubscription(projectId);
  const usage = await getUsage(projectId, metric);
  const limit = getTierLimit(subscription.tier, metric);
  return usage < limit;
}
```
